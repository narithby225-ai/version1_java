## ğŸ’¾ Database Transaction Flow

### Sale Transaction:
```sql
BEGIN TRANSACTION;

-- Step 1: Reduce stock
UPDATE products 
SET quantity = quantity - 5 
WHERE id = 1;

-- Step 2: Record sale
INSERT INTO sales 
(sale_date, product_name, quantity, total_price, 
 customer_name, customer_phone)
VALUES 
('2026-02-25', 'Nike Football', 5, 125.00, 
 'John Doe', '012345678');

COMMIT;
```

### Purchase Transaction:
```sql
BEGIN TRANSACTION;

-- Step 1: Log transaction
INSERT INTO stock_transactions 
(product_id, trans_type, quantity, unit_price, 
 total_amount, trans_date, note, user_name)
VALUES 
(1, 'PURCHASE', 100, 25.00, 2500.00, 
 '2026-02-25 10:30:00', 'Supplier ABC', 'admin');

-- Step 2: Increase stock
UPDATE products 
SET quantity = quantity + 100,
    price = 25.00
WHERE id = 1;

COMMIT;
```
